95  *
96  * Description: Program main function

103 int main(int argc, char *argv[])
104 {
105  MDIS_PATH path=0;
106  int32 n, count=0,incrTime=1;
107  int32 trigTime,testTime,setTime,getTime,status,shot;
108  char *device,*str,*errstr,buf[40];
109 
110  /*--------------------+
111  | check arguments |
112  +--------------------*/
113  if ((errstr = UTL_ILLIOPT("w=t=s=gio?", buf))) { /* check args */
114  printf("*** %s\n", errstr);
115  return(1);
116  }
117 
118  if (UTL_TSTOPT("?")) { /* help requested ? */
119  usage();
120  return(1);
121  }
122 
123  /*--------------------+
124  | get arguments |
125  +--------------------*/
126  for (device=NULL, n=1; n<argc; n++)
127  if (*argv[n] != '-') {
128  device = argv[n];
129  break;
130  }
131 
132  if ( (!device) || (argc < 3) ) {
133  usage();
134  return(1);
135  }
136 
137  trigTime = ((str = UTL_TSTOPT("w=")) ? atoi(str) : 0);
138  testTime = ((str = UTL_TSTOPT("t=")) ? atoi(str) : 0);
139  setTime = ((str = UTL_TSTOPT("s=")) ? atoi(str) : -1);
140  getTime = (UTL_TSTOPT("g") ? 1 : 0);
141  status = (UTL_TSTOPT("i") ? 1 : 0);
142  shot = (UTL_TSTOPT("o") ? 1 : 0);
143 
144  /*--------------------+
145  | open path |
146  +--------------------*/
147  if ((path = M_open(device)) < 0) {
148  PrintMdisError("open");
149  return(1);
150  }
151 
152  /*--------------------+
153  | get wdog-status |
154  +--------------------*/
155  if (status) {
156  if ((M_getstat(path, WDOG_STATUS, &status)) < 0) {
157  PrintMdisError("getstat WDOG_STATUS");
158  goto abort;
159  }
160  printf("Watchdog is %s\n", status ? "active" : "inactive");
161  }
162 
163  /*--------------------+
164  | get shot info |
165  +--------------------*/
166  if (shot) {
167  if ((M_getstat(path, WDOG_SHOT, &shot)) < 0) {
168  PrintMdisError("getstat WDOG_SHOT");
169  goto abort;
170  }
171  printf("Did the watchdog initiates the last system reset ?\n");
172  switch (shot) {
173  case 0: printf(" NO\n"); break;
174  case 1: printf(" YES\n"); break;
175  case 255: printf(" NOT IDENTIFIABLE\n"); break;
176  default : printf(" *** ERROR: DRIVER RETURNS AN ILLEGAL VALUE\n"); break;
177  }
178  }
179 
180  /*--------------------+
181  | get wdog-time |
182  +--------------------*/
183  if (getTime) {
184  if ((M_getstat(path, WDOG_TIME, &getTime)) < 0) {
185  PrintMdisError("getstat WDOG_TIME");
186  goto abort;
187  }
188  printf("Watchdog time: %dmsec\n", getTime);
189  }
190 
191  /*--------------------+
192  | set wdog-time |
193  +--------------------*/
194  if (0 < setTime) {
195  if ((M_setstat(path, WDOG_TIME, setTime)) < 0) {
196  PrintMdisError("setstat WDOG_TIME");
197  goto abort;
198  }
199  printf("Watchdog time set to %dmsec\n", setTime);
200  }
201 
202  /*--------------------+
203  | stop watchdog |
204  +--------------------*/
205  if (0 == setTime) {
206  if ((M_setstat(path, WDOG_STOP, 0)) < 0) {
207  PrintMdisError("setstat WDOG_STOP");
208  goto abort;
209  }
210  printf("Watchdog stopped\n");
211  }
212 
213  /*--------------------+
214  | watch |
215  +--------------------*/
216  if (trigTime) {
217  /* start watchdog */
218  if ((M_setstat(path, WDOG_START, 0)) < 0) {
219  PrintMdisError("setstat WDOG_START");
220  goto abort;
221  }
222  printf("Watchdog started - trigger all %dmsec\n", trigTime);
223 
224  /* trigger loop */
225  do {
226  UOS_Delay(trigTime);
227  count++;
228  printf(" (%6d) Trigger watchdog - Press any key to abort\n", count);
229  if ((M_setstat(path, WDOG_TRIG, 0)) < 0) {
230  PrintMdisError("setstat WDOG_TRIG");
231  goto abort;
232  }
233  } while(UOS_KeyPressed() == -1);
234 
235  /* try to stop watchdog */
236  if ((M_setstat(path, WDOG_STOP, 0)) < 0) {
237  PrintMdisError("setstat WDOG_STOP");
238  goto abort;
239  }
240  printf("Watchdog stopped\n");
241  }
242 
243  /*--------------------+
244  | test wdog-time |
245  +--------------------*/
246  if (testTime) {
247  /* start watchdog */
248  if ((M_setstat(path, WDOG_START, 0)) < 0) {
249  PrintMdisError("setstat WDOG_START");
250  goto abort;
251  }
252  printf("Watchdog started - force system reset\n", trigTime);
253 
254  /* trigger loop */
255  do {
256  UOS_Delay(testTime);
257  printf(" Trigger watchdog after %6dmsec\n", testTime);
258  if ((M_setstat(path, WDOG_TRIG, 0)) < 0) {
259  PrintMdisError("setstat WDOG_TRIG");
260  goto abort;
261  }
262  if (testTime >= 100) incrTime = 10;
263  if (testTime >= 1000) incrTime = 100;
264  if (testTime >= 10000) incrTime = 1000;
265  testTime += incrTime;
266  } while(1);
267  }
268 
269  /*--------------------+
270  | cleanup |
271  +--------------------*/
272  abort:
273  if (M_close(path) < 0)
274  PrintMdisError("close");
275 
276  return(0);
277 }
