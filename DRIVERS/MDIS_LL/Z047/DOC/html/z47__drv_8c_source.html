154 static int32 Z47_Init(
155  DESC_SPEC *descP,
156  OSS_HANDLE *osHdl,
157  MACCESS *ma,
158  OSS_SEM_HANDLE *devSemHdl,
159  OSS_IRQ_HANDLE *irqHdl,
160  LL_HANDLE **llHdlP
161 )
162 {
163  LL_HANDLE *llHdl = NULL;
164  u_int32 gotsize;
165  int32 error;
166  u_int32 value;
167 
168  /*------------------------------+
169  | prepare the handle |
170  +------------------------------*/
171  *llHdlP = NULL; /* set low-level driver handle to NULL */
172 
173  /* alloc */
174  if ((llHdl = (LL_HANDLE*)OSS_MemGet(
175  osHdl, sizeof(LL_HANDLE), &gotsize)) == NULL)
176  return (ERR_OSS_MEM_ALLOC);
177 
178  /* clear */
179  OSS_MemFill(osHdl, gotsize, (char*)llHdl, 0x00);
180 
181  /* init */
182  llHdl->memAlloc = gotsize;
183  llHdl->osHdl = osHdl;
184  llHdl->irqHdl = irqHdl;
185  llHdl->ma = *ma;
186 
187  /*------------------------------+
188  | init id function table |
189  +------------------------------*/
190  /* driver's ident function */
191  llHdl->idFuncTbl.idCall[0].identCall = Ident;
192  /* library's ident functions */
193  llHdl->idFuncTbl.idCall[1].identCall = DESC_Ident;
194  llHdl->idFuncTbl.idCall[2].identCall = OSS_Ident;
195  /* terminator */
196  llHdl->idFuncTbl.idCall[3].identCall = NULL;
197 
198  /*------------------------------+
199  | prepare debugging |
200  +------------------------------*/
201  DBG_MYLEVEL = OSS_DBG_DEFAULT; /* set OS specific debug level */
202  DBGINIT((NULL,&DBH));
203 
204  /*------------------------------+
205  | scan descriptor |
206  +------------------------------*/
207  if ((error = DESC_Init(descP, osHdl, &llHdl->descHdl)))
208  return (Cleanup(llHdl, error));
209 
210  /* DEBUG_LEVEL_DESC */
211  if ((error = DESC_GetUInt32(llHdl->descHdl, OSS_DBG_DEFAULT,
212  &value, "DEBUG_LEVEL_DESC")) &&
213  error != ERR_DESC_KEY_NOTFOUND)
214  return (Cleanup(llHdl, error));
215 
216  DESC_DbgLevelSet(llHdl->descHdl, value);
217 
218  /* DEBUG_LEVEL */
219  if ((error = DESC_GetUInt32(llHdl->descHdl, OSS_DBG_DEFAULT,
220  &llHdl->dbgLevel, "DEBUG_LEVEL")) &&
221  error != ERR_DESC_KEY_NOTFOUND)
222  return (Cleanup(llHdl, error));
223 
224  /*------------------------------+
225  | init hardware |
226  +------------------------------*/
227  /* Reset internal counter, WDOG_TIMEOUT, WDOG_IRQ signals
228  and restart watchdog functionality. Watchdog start/stop is not affected. */
229  MSETMASK_D32(llHdl->ma, Z47_CFG, Z47_CFG_WDOG_RESTART);
230 
231  *llHdlP = llHdl; /* set low-level driver handle */
232 
233  return (ERR_SUCCESS);
234 }

326 static int32 Z47_SetStat(
327  LL_HANDLE *llHdl,
328  int32 code,
329  int32 ch,
330  INT32_OR_64 value32_or_64
331 )
332 {
333  char *func = "LL - Z47_SetStat";
334  int32 value = (int32)value32_or_64; /* 32bit value */
335  MACCESS ma = llHdl->ma;
336  int32 error = ERR_SUCCESS;
337 
338  DBGWRT_1((DBH, "%s: ch=%d code=0x%04x value=0x%x\n",
339  func, ch, code, value));
340 
341  switch (code) {
342  /*--------------------------+
343  | debug level |
344  +--------------------------*/
345  case M_LL_DEBUG_LEVEL:
346  llHdl->dbgLevel = value;
347  break;
348  /*--------------------------+
349  | enable interrupts |
350  +--------------------------*/
351  case M_MK_IRQ_ENABLE:
352  break;
353  /*--------------------------+
354  | set irq counter |
355  +--------------------------*/
356  case M_LL_IRQ_COUNT:
357  llHdl->irqCount = value;
358  break;
359  /*--------------------------+
360  | channel direction |
361  +--------------------------*/
362  case M_LL_CH_DIR:
363  if (value != M_CH_INOUT) {
364  error = ERR_LL_ILL_DIR;
365  }
366  break;
367  /*--------------------------+
368  | old WDOG codes |
369  +--------------------------*/
370  case WDOG_START:
371  MSETMASK_D32(ma, Z47_CFG, Z47_CFG_WDOG_EN);
372  llHdl->wdStart = 1;
373  break;
374 
375  case WDOG_STOP:
376  MCLRMASK_D32(ma, Z47_CFG, Z47_CFG_WDOG_EN);
377  llHdl->wdStart = 0;
378  break;
379 
380  case WDOG_TRIG:
381  error = WdogTrig(llHdl, 0);
382  break;
383 
384  case WDOG_TIME:
385  /* given time [ms], IP core time [us] */
386  MWRITE_D32(ma, Z47_MAXT, value * 1000);
387  break;
388 
389  /*--------------------------+
390  | new WDOG codes |
391  +--------------------------*/
392  case WDOG_RESET_CTRL:
393  MSETMASK_D32(ma, Z47_CFG, Z47_CFG_WDOG_RESTART);
394  break;
395 
396  case WDOG_TRIG_PAT:
397  error = WdogTrig(llHdl, value);
398  break;
399 
400  case WDOG_TIME_MIN:
401  /* given time [us], IP core time [us] */
402  MWRITE_D32(ma, Z47_MINT, value);
403  break;
404 
405  case WDOG_TIME_MAX:
406  /* given time [us], IP core time [us] */
407  MWRITE_D32(ma, Z47_MAXT, value);
408  break;
409 
410  case WDOG_TIME_IRQ:
411  /* given time [us], IP core time [us] */
412  MWRITE_D32(ma, Z47_IRQT, value);
413  break;
414 
415  case WDOG_OUT_PIN:
416  switch (value){
417  case 0:
418  MCLRMASK_D32(ma, Z47_STAT, Z47_STAT_SET_WDOG_TOUT);
419  break;
420  case 1:
421  MSETMASK_D32(ma, Z47_STAT, Z47_STAT_SET_WDOG_TOUT);
422  break;
423  default:
424  DBGWRT_ERR((DBH, "*** %s(WDOG_OUT_PIN): illegal pattern 0x%x\n",
425  func, value));
426  return ERR_LL_ILL_PARAM;
427  }
428  break;
429 
430  case WDOG_OUT_REASON:
431  MCLRMASK_D32(ma, Z47_STAT, Z47_STAT_REASON_TOUT_MASK);
432  break;
433 
434  case WDOG_IRQ_PIN:
435  switch (value) {
436  case 0:
437  MCLRMASK_D32(ma, Z47_STAT, Z47_STAT_SET_WDOG_IRQ);
438  break;
439  case 1:
440  MSETMASK_D32(ma, Z47_STAT, Z47_STAT_SET_WDOG_IRQ);
441  break;
442  default:
443  DBGWRT_ERR((DBH, "*** %s(WDOG_IRQ_PIN): illegal pattern 0x%x\n",
444  func, value));
445  return ERR_LL_ILL_PARAM;
446  }
447  break;
448 
449  case WDOG_IRQ_SIGSET:
450  /* signal already installed ? */
451  if (llHdl->sigHdl) {
452  error = ERR_OSS_SIG_SET;
453  break;
454  }
455  error = OSS_SigCreate(OSH, value, &llHdl->sigHdl);
456  break;
457 
458  case WDOG_IRQ_SIGCLR:
459  /* signal not installed ? */
460  if (llHdl->sigHdl == NULL) {
461  error = ERR_OSS_SIG_CLR;
462  break;
463  }
464  error = OSS_SigRemove(OSH, &llHdl->sigHdl);
465  break;
466 
467  case WDOG_IRQ_REASON:
468  MCLRMASK_D32(ma, Z47_STAT, Z47_STAT_REASON_IRQ_MASK);
469  break;
470 
471  case WDOG_ERR_PIN:
472  switch (value) {
473  case 0:
474  MCLRMASK_D32(ma, Z47_STAT, Z47_STAT_SW_ERR);
475  break;
476  case 1:
477  MSETMASK_D32(ma, Z47_STAT, Z47_STAT_SW_ERR);
478  break;
479  default:
480  DBGWRT_ERR((DBH, "*** %s(WDOG_ERR_PIN): illegal pattern 0x%x\n",
481  func, value));
482  return ERR_LL_ILL_PARAM;
483  }
484  break;
485 
486  /*--------------------------+
487  | (unknown) |
488  +--------------------------*/
489  default:
490  error = ERR_LL_UNK_CODE;
491  }
492 
493  return (error);
494 }

513 static int32 Z47_GetStat(

524  int32 read;
525 
526  DBGWRT_1((DBH, "LL - Z47_GetStat: ch=%d code=0x%04x\n", ch, code));
527 
528  switch (code) {
529  /*--------------------------+
530  | debug level |
531  +--------------------------*/
532  case M_LL_DEBUG_LEVEL:
533  *valueP = llHdl->dbgLevel;
534  break;
535  /*--------------------------+
536  | number of channels |
537  +--------------------------*/
538  case M_LL_CH_NUMBER:
539  *valueP = CH_NUMBER;
540  break;
541  /*--------------------------+
542  | channel direction |
543  +--------------------------*/
544  case M_LL_CH_DIR:
545  *valueP = M_CH_INOUT;
546  break;
547  /*--------------------------+
548  | channel length [bits] |
549  +--------------------------*/
550  case M_LL_CH_LEN:
551  *valueP = 1;
552  break;
553  /*--------------------------+
554  | channel type info |
555  +--------------------------*/
556  case M_LL_CH_TYP:
557  *valueP = M_CH_BINARY;
558  break;
559  /*--------------------------+
560  | irq counter |
561  +--------------------------*/
562  case M_LL_IRQ_COUNT:
563  *valueP = llHdl->irqCount;
564  break;
565  /*--------------------------+
566  | ID PROM check enabled |
567  +--------------------------*/
568  case M_LL_ID_CHECK:
569  *valueP = 0;
570  break;
571  /*--------------------------+
572  | ident table pointer |
573  | (treat as non-block!) |
574  +--------------------------*/
575  case M_MK_BLK_REV_ID:
576  *value64P = (INT32_OR_64)&llHdl->idFuncTbl;
577  break;
578 
579  /*--------------------------+
580  | old WDOG codes |
581  +--------------------------*/
582  case WDOG_TIME:
583  /* IP core time [us], requested time [ms] */
584  *value64P = (MREAD_D32(ma, Z47_MAXT)) / 1000;
585  break;
586 
587  case WDOG_STATUS:
588  read = MREAD_D32(ma, Z47_STAT);
589  *value64P = (read & Z47_STAT_WDOG_CNT_STATUS) ? 1 : 0;
590  break;
591 
592  case WDOG_SHOT:
593  read = MREAD_D32(ma, Z47_STAT);
594  read = (read & Z47_STAT_REASON_TOUT_MASK) >> Z47_STAT_REASON_TOUT_SHIFT;
595  *value64P = ((read == Z47_REASON_MIN_TOUT) ||
596  (read == Z47_REASON_MAX_TOUT) ) ? 1 : 0;
597  break;
598 
599  /*--------------------------+
600  | new WDOG codes |
601  +--------------------------*/
602  case WDOG_TIME_MIN:
603  /* IP core time [us], requested time [us] */
604  *value64P = MREAD_D32(ma, Z47_MINT);
605  break;
606 
607  case WDOG_TIME_MAX:

610  break;
611 
612  case WDOG_TIME_IRQ:

620  break;

630  break;

955  return ERR_LL_ILL_PARAM;

963  return ERR_LL_ILL_PARAM;
964  }

966  val = pat;
967  }
968  /* no pattern given */
969  else {

971  if (read == 0)
972  read = WDOG_TRIGPAT(0);
973

975  val = (read & 0x0f) << 4;
976  val |= (read & 0xf0) >> 4;
977  }

979  /* write trigger pattern */
980  MWRITE_D32(llHdl->ma, Z47_CLR, val);
